New body (per-page loop):
```python
for page_path in pages:
    pid = page_id(page_path, wiki_dir)
    if pid == target_page_id:
        continue  # self — no lock
    # Fast-path: cheap pre-lock read + decision.
    try:
        content = page_path.read_text(encoding="utf-8")
    except (OSError, UnicodeDecodeError):
        continue
    fm_match = _FRONTMATTER_RE.match(content)
    if fm_match:
        frontmatter_section = fm_match.group(1)
        body = fm_match.group(2)
    else:
        frontmatter_section = ""
        body = content
    existing_links = extract_wikilinks(body)
    if target_page_id in existing_links:
        continue  # already linked — no lock
    # Peek: would the title match in masked body?
    _peek_body, _peek_masked, _peek_prefix = _mask_code_blocks(body)
    if not pattern.search(_peek_body):
        continue  # no match — no lock (fast-path)

    # Under-lock RMW path.
    try:
        with file_lock(page_path, timeout=_INJECT_LOCK_TIMEOUT):
            # Re-read under lock to catch concurrent writer.
            try:
                content = page_path.read_text(encoding="utf-8")
            except (OSError, UnicodeDecodeError):
                continue
            fm_match = _FRONTMATTER_RE.match(content)
            if fm_match:
                frontmatter_section = fm_match.group(1)
                body = fm_match.group(2)
            else:
                frontmatter_section = ""
                body = content
            existing_links = extract_wikilinks(body)
            if target_page_id in existing_links:
                continue  # race lost — concurrent injector won
            original_body = body
            body, masked_code, mask_prefix = _mask_code_blocks(body)
            if not pattern.search(body):
                continue  # race lost
            safe_title = wikilink_display_escape(title)
            replacement = f"[[{target_page_id}|{safe_title}]]"
            new_body = body
            for match in pattern.finditer(body):
                start = match.start()
                result = _check_not_in_wikilink(match, body, pid, replacement)
                if result is None:
                    continue
                new_body = body[:start] + replacement + body[match.end() :]
                break
            new_body = _unmask_code_blocks(new_body, masked_code, mask_prefix)
            if new_body != original_body:
                atomic_text_write(frontmatter_section + new_body, page_path)
                updated.append(pid)
                logger.info("Injected wikilink to %s in %s", target_page_id, pid)
    except TimeoutError:
        logger.warning(
            "Skipping inject_wikilinks on %s: lock timeout after %.2fs",
            pid,
            _INJECT_LOCK_TIMEOUT,
        )
        continue
```

Note: `continue` inside a `with` statement exits the `with` (lock released) then continues the outer loop — idiomatic Python, safe. **Verify**: `file_lock` raises `TimeoutError` on timeout (check `io.py:252` implementation). If it raises a different type, catch that instead.

### 5a — `_emit_ingest_jsonl` helper (AC11 + AC12 + AC13)

Add near the top of `pipeline.py` (after existing imports):
```python
import json
import uuid

_INGEST_JSONL_STAGES = frozenset({"start", "duplicate_skip", "success", "failure"})
_INGEST_JSONL_ERROR_SUMMARY_MAX = 2048


def _emit_ingest_jsonl(
    stage: str,
    request_id: str,
    source_ref: str,
    source_hash: str,
    outcome: dict,
) -> None:
    """Append one JSON row to <PROJECT_ROOT>/.data/ingest_log.jsonl.

    Best-effort (Q8): OSError is logged at WARNING and swallowed so JSONL write
    failure never masks the ingest outcome.
    Schema (Q19): scalar args + stage enum validation + explicit field allowlist.
    Durability (Q9): file_lock + fsync on every append.
    Rotation (AC12): rotate_if_oversized called INSIDE the same file_lock.
    Redaction (AC13): `outcome["error_summary"]` is redacted + truncated.
    """
    if stage not in _INGEST_JSONL_STAGES:
        raise ValueError(
            f"Unknown ingest_jsonl stage: {stage!r}; expected one of {_INGEST_JSONL_STAGES}"
        )
    # [Plan-gate amendment 2] — enforce explicit field allowlist (T1). A `dict(outcome)`
    # copy leaks arbitrary caller fields into the audit log.
    _ALLOWED_OUTCOME_KEYS = {
        "pages_created",
        "pages_updated",
        "pages_skipped",
        "error_summary",
    }
    unexpected = set(outcome) - _ALLOWED_OUTCOME_KEYS
    if unexpected:
        logger.warning(
            "Dropping unexpected ingest_log outcome fields: %s",
            ", ".join(sorted(unexpected)),
        )
    safe_outcome: dict = {}
    for key in ("pages_created", "pages_updated", "pages_skipped"):
        if key in outcome:
            safe_outcome[key] = int(outcome[key])
    err = outcome.get("error_summary")
    if isinstance(err, str) and err:
        safe_outcome["error_summary"] = sanitize_text(err)[:_INGEST_JSONL_ERROR_SUMMARY_MAX]
    row = {
        "ts": datetime.now(UTC).isoformat().replace("+00:00", "Z"),
        "request_id": request_id,
        "source_ref": source_ref,
        "source_hash": source_hash,
        "stage": stage,
        "outcome": safe_outcome,
    }
    jsonl_path = PROJECT_ROOT / ".data" / "ingest_log.jsonl"
    try:
        jsonl_path.parent.mkdir(parents=True, exist_ok=True)
        with file_lock(jsonl_path):
            rotate_if_oversized(jsonl_path, LOG_SIZE_WARNING_BYTES, "ingest_log")
            with jsonl_path.open("a", encoding="utf-8", newline="\n") as f:
                f.write(json.dumps(row, ensure_ascii=False) + "\n")
                f.flush()
                os.fsync(f.fileno())
    except OSError as e:
        logger.warning("Failed to write ingest_log.jsonl (best-effort): %s", e)
```
