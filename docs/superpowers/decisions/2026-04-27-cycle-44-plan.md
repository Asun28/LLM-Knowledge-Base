```python
def atomic_text_write(content: str, path: Path | str, *, exclusive: bool = False) -> None:
    """Atomic create-or-replace text write.

    exclusive=True: O_CREAT|O_EXCL|O_WRONLY for slug-collision detection
                    (raises FileExistsError on conflict). Combines O_EXCL with
                    tempfile+rename for crash-safety.
    exclusive=False (default): tempfile + os.replace for crash-atomic replace.
    """
    path = Path(path)
    if exclusive:
        # Race-safe reservation via O_EXCL, then crash-safe write via tempfile+rename
        fd = os.open(str(path), os.O_CREAT | os.O_EXCL | os.O_WRONLY, 0o644)
        os.close(fd)
        try:
            _write_atomically(content, path)  # existing tempfile+rename body
        except BaseException:
            path.unlink(missing_ok=True)
            raise
    else:
        _write_atomically(content, path)
```
