```python
def refine_page(...) -> dict:
    # ... existing path traversal + exists checks unchanged ...

    import uuid
    attempt_id = uuid.uuid4().hex[:8]
    timestamp = datetime.now().isoformat(timespec="seconds")

    # Lock nest: page_lock OUTER, history_lock INNER (single span covers pending + page + flip).
    # See "Concurrency notes" in module docstring: page_path FIRST, history_path SECOND.
    with file_lock(page_path):
        with file_lock(resolved_history_path):
            # Phase 1 — write pending row BEFORE page.
            history = load_review_history(resolved_history_path)
            pending_entry = {
                "timestamp": timestamp,
                "page_id": page_id,
                "revision_notes": revision_notes,
                "content_length": len(updated_content),
                "status": "pending",
                "attempt_id": attempt_id,
            }
            history.append(pending_entry)
            if len(history) > MAX_REVIEW_HISTORY_ENTRIES:
                history = history[-MAX_REVIEW_HISTORY_ENTRIES:]
            save_review_history(history, resolved_history_path)

            # Phase 2 — write page atomically.
            try:
                # ... existing frontmatter parse, yaml validate, body write ...
                atomic_text_write(new_text, page_path)
            except OSError as e:
                # Flip pending → failed under SAME lock span.
                history = load_review_history(resolved_history_path)
                for row in history:
                    if row.get("attempt_id") == attempt_id:
                        row["status"] = "failed"
                        row["error"] = str(e)
                        break
                save_review_history(history, resolved_history_path)
                return {"error": f"Failed to write page {page_id}: {e}"}

            # Phase 3 — flip pending → applied under SAME lock span.
            history = load_review_history(resolved_history_path)
            for row in history:
                if row.get("attempt_id") == attempt_id:
                    row["status"] = "applied"
                    break
            save_review_history(history, resolved_history_path)

    # Append to wiki/log.md (best-effort outside locks — existing pattern).
    # ...

    return {"page_id": page_id, "updated": True, "revision_notes": revision_notes}


def list_stale_pending(
    hours: int = 24,
    *,
    history_path: Path | None = None,
) -> list[dict]:
    """Return history entries with status='pending' older than `hours`.

    Pure read; no locks. Intended for operator visibility on processes that
    crashed mid-refine (pending rows that never flipped to applied/failed).
    A full sweep tool (auto-promote or delete) is deferred to a future cycle.
    """
    from datetime import datetime, timedelta
    cutoff = datetime.now() - timedelta(hours=hours)
    history = load_review_history(history_path)
    stale = []
    for row in history:
        if row.get("status") != "pending":
            continue
        try:
            ts = datetime.fromisoformat(row.get("timestamp", ""))
        except (ValueError, TypeError):
            continue
        if ts < cutoff:
            stale.append(row)
    return stale
```
