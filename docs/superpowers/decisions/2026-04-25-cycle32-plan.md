    def test_fair_queue_stagger_probabilistic_ordering(self, tmp_path):
        """AC7 — probabilistic fairness mitigation under N=3 thundering-herd.
        Uses threading.Barrier for simultaneous entry; 10 trials, 80% tolerance
        to OS scheduler jitter per C6."""
        import time
        from threading import Barrier, Thread

        from kb.utils.io import file_lock

        lock_path = tmp_path / "fairness.lock"
        n_workers = 3
        trials = 10
        first_entrant_won = 0

        for _ in range(trials):
            acquire_time_by_entry: dict[int, float] = {}
            barrier = Barrier(n_workers)

            def worker() -> None:
                entry_idx = barrier.wait()  # returns 0..n_workers-1 by entry order
                with file_lock(lock_path, timeout=5.0):
                    acquire_time_by_entry[entry_idx] = time.perf_counter()

            threads = [Thread(target=worker) for _ in range(n_workers)]
            for t in threads:
                t.start()
            for t in threads:
                t.join()

            first = acquire_time_by_entry.get(0, float("inf"))
            last = acquire_time_by_entry.get(n_workers - 1, 0.0)
            if first <= last:
                first_entrant_won += 1

        assert first_entrant_won >= 8, (
            f"Fair-queue stagger signal weak: {first_entrant_won}/10 trials "
            "showed first-entrant acquire ≤ last-entrant. Expected ≥8/10."
        )

def ingest_content(
    filename: str,
    source_type: str,
    content_file,
    extraction_json_file,
    url: str,
    use_api: bool,
) -> None:
    """One-shot ingest: save content to raw/ and create wiki pages.

    Cycle 32 AC4 — CLI parity for MCP `kb_ingest_content`. Content + extraction
    JSON forwarded verbatim; MCP tool is authoritative for validation.
    """
    import os as _os  # noqa: PLC0415

    from kb.config import MAX_INGEST_CONTENT_CHARS  # noqa: PLC0415
    from kb.mcp.core import kb_ingest_content  # noqa: PLC0415

    # Cycle 32 C13 — stat guard for --content-file BEFORE read.
    try:
        st_size = _os.fstat(content_file.fileno()).st_size
        if st_size > MAX_INGEST_CONTENT_CHARS:
            click.echo(
                f"Error: --content-file size {st_size} exceeds "
                f"{MAX_INGEST_CONTENT_CHARS}.",
                err=True,
            )
            sys.exit(1)
    except (OSError, AttributeError):
        pass  # stdin / non-seekable; defer to MCP cap.

    # Cycle 32 C13 — stat guard for --extraction-json-file (same cap per Q9).
    if extraction_json_file is not None:
        try:
            ej_size = _os.fstat(extraction_json_file.fileno()).st_size
            if ej_size > MAX_INGEST_CONTENT_CHARS:
                click.echo(
                    f"Error: --extraction-json-file size {ej_size} exceeds "
                    f"{MAX_INGEST_CONTENT_CHARS}.",
                    err=True,
                )
                sys.exit(1)
        except (OSError, AttributeError):
            pass

    content = content_file.read()
    extraction_json = extraction_json_file.read() if extraction_json_file else ""

    try:
        output = kb_ingest_content(
            content=content,
            filename=filename,
            source_type=source_type,
            extraction_json=extraction_json,
            url=url,
            use_api=use_api,
        )
        if _is_mcp_error_response(output):
            click.echo(output, err=True)
            sys.exit(1)
        click.echo(output)
    except Exception as exc:
        _error_exit(exc)
