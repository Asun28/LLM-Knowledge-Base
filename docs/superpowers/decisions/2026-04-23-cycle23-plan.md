     ```python
     def rebuild_indexes(
         wiki_dir: Path | None = None,
         *,
         hash_manifest: Path | None = None,
         vector_db: Path | None = None,
     ) -> dict:
         """Wipe derived indices so the next compile/query pass regenerates from source.

         Clears: HASH_MANIFEST, vector index DB (under <wiki_dir>.parent/.data/),
         and in-process LRU caches in kb.ingest.extractors / kb.utils.pages.

         Acquires file_lock(HASH_MANIFEST, timeout=1.0) to avoid racing a
         concurrent compile_wiki save. Audit-appends to wiki/log.md.

         Args:
             wiki_dir: Target wiki; defaults to WIKI_DIR. Must resolve under PROJECT_ROOT.
             hash_manifest: Override manifest path (default: HASH_MANIFEST).
             vector_db: Override vector DB path (default: derived from wiki_dir).

         Returns:
             {
               "manifest": {"cleared": bool, "error": str | None},
               "vector": {"cleared": bool, "error": str | None},
               "caches_cleared": list[str],
               "audit_written": bool,
             }

         Raises:
             ValidationError: wiki_dir outside PROJECT_ROOT.
         """
     ```
   - Implementation sketch (inside body):
     ```python
     from kb.config import PROJECT_ROOT, WIKI_DIR
     from kb.errors import ValidationError
     from kb.utils.io import file_lock
     from kb.utils.wiki_log import append_wiki_log

     effective_wiki = Path(wiki_dir).expanduser() if wiki_dir else WIKI_DIR
     root_resolved = PROJECT_ROOT.resolve()
     wiki_resolved = effective_wiki.resolve()
     if not (wiki_resolved == root_resolved or wiki_resolved.is_relative_to(root_resolved)):
         raise ValidationError(f"wiki_dir must be inside project root")
     if effective_wiki.is_absolute() and not (effective_wiki == root_resolved or effective_wiki.is_relative_to(root_resolved)):
         raise ValidationError(f"wiki_dir must be inside project root")

     manifest_path = hash_manifest or HASH_MANIFEST
     from kb.query.embeddings import _vector_index_path
     vector_path = vector_db or _vector_index_path(effective_wiki)

     result = {
         "manifest": {"cleared": False, "error": None},
         "vector": {"cleared": False, "error": None},
         "caches_cleared": [],
         "audit_written": False,
     }

     # Manifest: locked unlink
     try:
         with file_lock(manifest_path, timeout=1.0):
             if manifest_path.exists():
                 manifest_path.unlink()
                 result["manifest"]["cleared"] = True
             else:
                 result["manifest"]["cleared"] = True  # nothing to clear is OK
     except TimeoutError:
         result["manifest"]["error"] = "lock busy"
     except OSError as e:
         result["manifest"]["error"] = str(e)

     # Vector DB: unlocked unlink (single-writer contract; embeddings.py has its own lock)
     try:
         if vector_path.exists():
             vector_path.unlink()
             result["vector"]["cleared"] = True
         else:
             result["vector"]["cleared"] = True
     except OSError as e:
         result["vector"]["error"] = str(e)

     # Caches
     from kb.ingest.extractors import clear_template_cache
     from kb.utils.pages import load_page_frontmatter, load_purpose
     clear_template_cache()
     result["caches_cleared"].append("kb.ingest.extractors.clear_template_cache")
     try:
         load_page_frontmatter.cache_clear()
         result["caches_cleared"].append("kb.utils.pages.load_page_frontmatter")
     except AttributeError:
         pass
     try:
         load_purpose.cache_clear()
         result["caches_cleared"].append("kb.utils.pages.load_purpose")
     except AttributeError:
         pass

     # Audit
     log_path = effective_wiki / "log.md"
     msg = (
         f"manifest={result['manifest']['cleared']} "
         f"vector={result['vector']['cleared']} "
         f"caches={len(result['caches_cleared'])}"
     )
     try:
         append_wiki_log("rebuild-indexes", msg, log_path)
         result["audit_written"] = True
     except OSError as e:
         logger.warning("rebuild_indexes: audit write failed: %s", e)

     return result
     ```

### TASK 5 — Cross-process file_lock regression (Group C, AC7)

**Files:**
- `tests/test_cycle23_file_lock_multiprocessing.py` (new, `@pytest.mark.integration`)

**Change:**
```python
"""Cross-process file_lock regression — spawn child that holds the lock,
verify parent TimeoutError + recovery after child release.

Phase 4.5 HIGH-Deferred. Uses multiprocessing 'spawn' context unconditionally
so semantics match across Windows / Linux / macOS.
"""
from __future__ import annotations

import multiprocessing as mp
import time
from pathlib import Path

import pytest

from kb.utils.io import file_lock


def _child_hold_lock(lock_path_str: str, pid_sentinel_str: str, hold_seconds: float):
    """Top-level worker — must be picklable for spawn.

    Acquires file_lock, writes child PID to sentinel, sleeps hold_seconds, releases.
    """
    import os
    from pathlib import Path
    from kb.utils.io import file_lock

    lock_path = Path(lock_path_str)
    pid_sentinel = Path(pid_sentinel_str)
    with file_lock(lock_path, timeout=5.0):
        pid_sentinel.write_text(str(os.getpid()), encoding="ascii")
        time.sleep(hold_seconds)


@pytest.mark.integration
def test_cross_process_file_lock_timeout_then_recovery(tmp_path):
    lock_target = tmp_path / "target.json"
    lock_target.write_text("{}", encoding="utf-8")
    pid_sentinel = tmp_path / "child.pid"

    ctx = mp.get_context("spawn")
    p = ctx.Process(target=_child_hold_lock, args=(str(lock_target), str(pid_sentinel), 3.0))
    p.start()
    try:
        # Wait up to 5s for child to grab the lock + write PID
        deadline = time.monotonic() + 5.0
        while not pid_sentinel.exists() and time.monotonic() < deadline:
            time.sleep(0.05)
        assert pid_sentinel.exists(), "child did not acquire lock in time"
        assert int(pid_sentinel.read_text(encoding="ascii")) == p.pid

        # Parent attempts acquire with short timeout — child still holding
        with pytest.raises(TimeoutError):
            with file_lock(lock_target, timeout=0.5):
                pytest.fail("unexpectedly acquired lock while child held it")
    finally:
        p.join(timeout=10)
        if p.is_alive():
            p.kill()
            p.join(timeout=2)

    # Child released; parent should now succeed
    with file_lock(lock_target, timeout=1.0):
        lock_target.write_text("parent-wrote", encoding="utf-8")
    assert lock_target.read_text(encoding="utf-8") == "parent-wrote"
```
