    """CLI ingest --type choices must match SOURCE_TYPE_DIRS."""

    def test_all_source_types_available(self):

        from kb.config import SOURCE_TYPE_DIRS

        for key in SOURCE_TYPE_DIRS:
            assert key in choices, f"Source type '{key}' missing from CLI choices"

### Step 2: Fix cli.py — compile exit code

- [ ] **2a: Add exit code 1 on compile errors**

In `src/kb/cli.py`, change the compile command (lines 77-81):

```python
# OLD:
        if result["errors"]:
            click.echo(f"  Errors: {len(result['errors'])}")
            for err in result["errors"]:
                click.echo(f"    ! {err['source']}: {err['error']}", err=True)
        click.echo("Done.")
# NEW:
        if result["errors"]:
            click.echo(f"  Errors: {len(result['errors'])}")
            for err in result["errors"]:
                click.echo(f"    ! {err['source']}: {err['error']}", err=True)
            ctx = click.get_current_context()
            ctx.exit(1)
        click.echo("Done.")
```

Note: `ctx.exit(1)` is Click-idiomatic. The `@click.pass_context` decorator is NOT needed —
`click.get_current_context()` retrieves the active context from any Click callback.
