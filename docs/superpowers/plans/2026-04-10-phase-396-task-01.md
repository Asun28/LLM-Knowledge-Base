```python
def yaml_escape(value: str) -> str:
    """Escape a string for safe YAML double-quote style.

    Handles backslashes, double quotes, newlines, tabs, carriage returns, null bytes,
    and ASCII control characters forbidden by the YAML spec.
    """
    if "\0" in value:
        logger.warning("Null byte removed from YAML value (possible data corruption)")
        value = value.replace("\0", "")
    # Strip ASCII control chars forbidden in YAML double-quoted scalars
    value = re.sub(r"[\x01-\x08\x0b\x0c\x0e-\x1f\x7f]", "", value)
    return (
        value.replace("\\", "\\\\")
        .replace('"', '\\"')
        .replace("\n", "\\n")
        .replace("\r", "\\r")
        .replace("\t", "\\t")
    )
```

def normalize_sources(sources: str | list | None) -> list[str]:
    """Normalize frontmatter 'source' field to always be a list of strings."""
    if sources is None:

    if isinstance(sources, str):
        return [sources]

def extract_raw_refs(text: str) -> list[str]:

    matches = re.findall(

```python
"""Content hashing for incremental compilation (hash-based change detection)."""

import hashlib
from pathlib import Path

_HASH_CHUNK_SIZE = 65536  # 64 KB


def content_hash(path: Path | str) -> str:
    """Compute first 32 hex chars (128-bit prefix of SHA-256) of a file's contents.

    Uses streaming reads to avoid loading large files entirely into memory.
    """
    path = Path(path) if isinstance(path, str) else path
    h = hashlib.sha256()
    with path.open("rb") as f:
        while chunk := f.read(_HASH_CHUNK_SIZE):
            h.update(chunk)
    return h.hexdigest()[:32]
```
