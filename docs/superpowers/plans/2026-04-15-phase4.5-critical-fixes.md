```python
@contextmanager
def file_lock(path: Path, timeout: float = 5.0):
    lock_path = path.with_suffix(path.suffix + ".lock")
    lock_path.parent.mkdir(parents=True, exist_ok=True)
    my_pid_bytes = str(os.getpid()).encode()
    deadline = time.monotonic() + timeout
    acquired = False
    try:
        while not acquired:
            try:
                fd = os.open(str(lock_path), os.O_CREAT | os.O_EXCL | os.O_WRONLY)
                try:
                    os.write(fd, my_pid_bytes)
                finally:
                    os.close(fd)
                acquired = True
            except (FileExistsError, PermissionError):
                if time.monotonic() > deadline:
                    # stale-lock logic unchanged — verify PID, maybe steal
                    try:
                        stale_pid = int(lock_path.read_text().strip())
                        os.kill(stale_pid, 0)
                        raise TimeoutError(
                            f"Lock {lock_path} held by running PID {stale_pid}. "
                            "Stop that process or delete the lock file."
                        )
                    except (ValueError, OSError):
                        pass
                    lock_path.unlink(missing_ok=True)
                    time.sleep(0.05)
                    continue
                time.sleep(0.05)
        yield
    finally:
        if acquired:
            lock_path.unlink(missing_ok=True)
```
